CREATE TABLE IF NOT EXISTS `sys`.`Student` (  `username` VARCHAR(16) NOT NULL,  `password` VARCHAR(32) NOT NULL,  `create_time` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,  `SID` INT(11) NOT NULL,  `name` VARCHAR(45) NULL DEFAULT NULL,  `FID` INT(11) NULL DEFAULT NULL,  PRIMARY KEY (`SID`),  INDEX `FID_idx` (`FID` ASC),  CONSTRAINT `FID`    FOREIGN KEY (`FID`)    REFERENCES `sys`.`Faculty` (`FID`)    ON DELETE NO ACTION    ON UPDATE NO ACTION)ENGINE = InnoDBDEFAULT CHARACTER SET = utf8;CREATE TABLE IF NOT EXISTS `sys`.`Faculty` (  `username` VARCHAR(16) NOT NULL,  `password` VARCHAR(32) NOT NULL,  `create_time` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,  `name` VARCHAR(45) NULL DEFAULT NULL,  `FID` INT(11) NOT NULL,  PRIMARY KEY (`FID`))ENGINE = InnoDBDEFAULT CHARACTER SET = utf8;CREATE TABLE IF NOT EXISTS `sys`.`Courses` (  `CID` INT(11) NOT NULL,  `name` VARCHAR(255) NOT NULL,  `FID` VARCHAR(45) NULL DEFAULT NULL,  PRIMARY KEY (`CID`),  CONSTRAINT `FID`    FOREIGN KEY (`CID`)    REFERENCES `sys`.`Faculty` (`FID`)    ON DELETE NO ACTION    ON UPDATE NO ACTION)ENGINE = InnoDBDEFAULT CHARACTER SET = utf8;CREATE TABLE IF NOT EXISTS `sys`.`Grades` (  `SID` INT(11) NOT NULL,  `CID` INT(11) NOT NULL,  `Grade` INT(11) NULL DEFAULT NULL,  `FID` INT(11) NULL DEFAULT NULL,  INDEX `FID_idx` (`FID` ASC),  INDEX `SID_idx` (`SID` ASC),  CONSTRAINT `FID`    FOREIGN KEY (`FID`)    REFERENCES `sys`.`Faculty` (`FID`)    ON DELETE NO ACTION    ON UPDATE NO ACTION,  CONSTRAINT `SID`    FOREIGN KEY (`SID`)    REFERENCES `sys`.`Student` (`SID`)    ON DELETE NO ACTION    ON UPDATE NO ACTION)ENGINE = InnoDBDEFAULT CHARACTER SET = utf8;DROP TABLE IF EXISTS `sys`.`sys_config` ;
